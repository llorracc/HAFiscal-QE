# Format: package-name # collection-membership
# Lines starting with # are comments and ignored
# Empty lines are ignored

# This file is used by:
#   - Dockerfile (Docker image builds)
#   - reproduce/docker/setup.sh (DevContainer setup)
#
# To add a new package:
#   1. Add it to this file with appropriate collection annotation
#   2. The scripts will automatically filter and install only standalone packages
